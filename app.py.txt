# PASTE YOUR GENERAL-ACCESS GOOGLE SHEET URL HERE:
SHEET_URL = "https://docs.google.com/spreadsheets/d/1IHtomumrXZrNv9nrn3q7wjT7M2Nj0Jj8TkasAH6kwrw/edit?usp=sharing"

@st.cache_data(ttl=15) # Refreshes and inspects your Google Sheet every 15 seconds
def fetch_clean_tab(base_url, sheet_name):
    try:
        # Re-engineers standard sheet link to query automated CSV exports cleanly
        csv_endpoint = base_url.split("/edit")[0] + f"/gviz/tq?tqx=out:csv&sheet={sheet_name}"
        df = pd.read_csv(csv_endpoint, skiprows=3) # Skips summary headers to parse raw data columns
        # Strip unnamed columns
        df = df.loc[:, ~df.columns.str.contains('^Unnamed')]
        return df
    except Exception as e:
        return pd.DataFrame()

# Stream your specific workbook matrices
df_summary = fetch_clean_tab(SHEET_URL, "Weekly Summary")
df_bifurcation = fetch_clean_tab(SHEET_URL, "Sales Bifurcation")
df_crr_team = fetch_clean_tab(SHEET_URL, "CRR Consolidated")
df_stock_interest = fetch_clean_tab(SHEET_URL, "Stock Interest")
df_prod = fetch_clean_tab(SHEET_URL, "Production OFD OTD")

if df_summary.empty:
    st.warning("⚠️ Access stream initializing... Please verify that your shared Google Sheet URL is securely pasted on Line 26.")
else:
    # Cleanup data streams to drop empty rows
    df_summary = df_summary[df_summary['Week'].str.contains('Week|Total', na=False, case=False)].reset_index(drop=True)
    df_w1 = df_summary[df_summary['Week'] == 'Week 1'].iloc[0]

    # Dashboard Banner Branding Header

    st.caption("Performance Activity Dashboard • Live Matrix Sync Engine")

    m1, m2, m3 = st.columns(3)
    
    with m1:
        val_crr = float(df_w1['CRR Actual ₹ Cr'])
        diff_crr = float(df_w1['CRR Diff'])
        st.metric(label="CRR Sales (Actual)", value=f"₹ {val_crr:.2f} Cr", 
                  delta=f"{diff_crr:.2f} Cr Shortfall" if diff_crr < 0 else f"▲ {diff_crr:.2f} Cr",
                  delta_color="normal" if diff_crr >= 0 else "inverse")
    with m2:
        val_nbd = float(df_w1['NBD Actual ₹ Cr'])
        diff_nbd = float(df_w1['NBD Diff'])
        st.metric(label="NBD Sales (Actual)", value=f"₹ {val_nbd:.2f} Cr", 
                  delta=f"{val_nbd:.2f} Cr Shortfall" if diff_nbd < 0 else f"▲ {diff_nbd:.2f} Cr",
                  delta_color="normal" if diff_nbd >= 0 else "inverse")
    with m3:
        val_coll = float(df_w1['Collection Actual ₹ Cr'])
        diff_coll = float(df_w1['Collection Diff'])
        st.metric(label="Financial Collections", value=f"₹ {val_coll:.2f} Cr",

    # Operational Sub-row KPIs

    m4, m5, m6 = st.columns(3)

        otd_pct = float(df_w1['OTD Actual %']) * 100 if float(df_w1['OTD Actual %']) <= 1 else float(df_w1['OTD Actual %'])
